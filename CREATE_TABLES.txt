CREATE COLUMN TABLE "U556414"."ALLSTATIONS"(
	"station_name" NVARCHAR(36) NOT NULL,
	"line_name" NVARCHAR(10) NOT NULL,
	"district_name" NVARCHAR(36),
	"station_link" NVARCHAR(100),
	"LATITUDE" DECIMAL(15, 8) CS_FIXED NOT NULL,
	"LONGITUDE" DECIMAL(15, 8) CS_FIXED NOT NULL
)

IMPORT FROM CSV FILE '/stationData/allStations.csv' INTO "U556414"."ALLSTATIONS" 
   WITH RECORD DELIMITED BY '\n'
   FIELD DELIMITED BY ',';

	"station_name" NVARCHAR(36) NOT NULL,
	"station_id" NVARCHAR(10) NOT NULL,
	"line_name" NVARCHAR(10) NOT NULL
)

IMPORT FROM CSV FILE '/stationData/allStations.csv' INTO "U556414"."SUBWAYLINES"
